  // ── Core calculation (returns result object) ──
  function computeSim(s) {
    const confirm  = s.confirm  / 100;
    const tauxLiv  = s.tauxLiv  / 100;
    const nbCmd    = s.nbCmd    || 1;
    const dollarDZD = s.dollarDay * s.dollar;

    const pubParCmd   = dollarDZD / nbCmd;
    const coutConf    = pubParCmd  / (confirm  || 1);
    const coutLivr    = coutConf   / (tauxLiv  || 1);
    const retourCmd   = s.retour * (1 - tauxLiv) / (tauxLiv || 1);
    const cmdLivrees  = nbCmd * confirm * tauxLiv;
    const benefice    = s.vente - s.achat - s.livraison - s.emballage - coutLivr - retourCmd - s.cod;
    const netDay      = benefice * cmdLivrees;
    const netMonth    = netDay * 30;
    const totalInv    = s.achat + s.emballage + s.livraison + coutLivr;
    const roi         = totalInv > 0 ? (benefice / totalInv) * 100 : 0;
    const roas        = pubParCmd > 0 ? s.vente / pubParCmd : 0;
    const cout        = s.dollarDay / nbCmd;

    return { pubParCmd, coutConf, coutLivr, retourCmd, cmdLivrees, benefice, netDay, netMonth, roi, roas, cout, nbCmd };
  }

  // ── Read form & update current sim in memory ──
  function readForm() {
    const s = simulations.find(x => x.id === currentSimId);
    if (!s) return;
    const mode = document.getElementById('f-calcmode').value;
    s.name      = document.getElementById('f-name').value.trim().toUpperCase() || s.name;
    s.type      = document.getElementById('f-type').value;
    s.achat     = parseFloat(document.getElementById('f-achat').value)      || 0;
    s.dollar    = parseFloat(document.getElementById('f-dollar').value)     || 1;
    s.vente     = parseFloat(document.getElementById('f-vente').value)      || 0;
    s.dollarDay = parseFloat(document.getElementById('f-dollars-day').value)|| 0;
    s.livraison = parseFloat(document.getElementById('f-livraison').value)  || 0;
    s.cod       = parseFloat(document.getElementById('f-cod').value)        || 0;
    s.emballage = parseFloat(document.getElementById('f-emballage').value)  || 0;
    s.retour    = parseFloat(document.getElementById('f-retour').value)     || 0;
    s.confirm   = parseFloat(document.getElementById('f-confirm').value)    || 0;
    s.tauxLiv   = parseFloat(document.getElementById('f-taux-liv').value)   || 0;
    if (mode === 'cout') {
      s.cout  = parseFloat(document.getElementById('f-cout').value) || 0;
      s.nbCmd = s.dollarDay / (s.cout || 1);
    } else {
      s.nbCmd = parseFloat(document.getElementById('f-commandes').value) || 0;
      s.cout  = s.nbCmd > 0 ? s.dollarDay / s.nbCmd : 0;
    }
  }

  // ── Main Calculation (reads form live) ──
  function calculate() {
    const achat     = parseFloat(document.getElementById('f-achat').value)       || 0;
    const vente     = parseFloat(document.getElementById('f-vente').value)       || 0;
    const dollar    = parseFloat(document.getElementById('f-dollar').value)      || 1;
    const dollarDay = parseFloat(document.getElementById('f-dollars-day').value) || 0;
    const livraison = parseFloat(document.getElementById('f-livraison').value)   || 0;
    const cod       = parseFloat(document.getElementById('f-cod').value)         || 0;
    const emballage = parseFloat(document.getElementById('f-emballage').value)   || 0;
    const retour    = parseFloat(document.getElementById('f-retour').value)      || 0;
    const confirm   = (parseFloat(document.getElementById('f-confirm').value)    || 0) / 100;
    const tauxLiv   = (parseFloat(document.getElementById('f-taux-liv').value)   || 0) / 100;

    const mode = document.getElementById('f-calcmode').value;
    let coutUSD = 0, nbCmd = 0;
    if (mode === 'cout') {
      coutUSD = parseFloat(document.getElementById('f-cout').value) || 0;
      nbCmd   = dollarDay / (coutUSD || 1);
    } else {
      nbCmd   = parseFloat(document.getElementById('f-commandes').value) || 0;
      coutUSD = nbCmd > 0 ? dollarDay / nbCmd : 0;
    }
    document.getElementById('f-commandes').value = Math.round(nbCmd * 100) / 100;
    document.getElementById('f-cout').value       = Math.round(coutUSD * 100) / 100;

    const pubParCmd   = (dollarDay * dollar) / (nbCmd || 1);
    const coutCmdConf = pubParCmd  / (confirm  || 1);
    const coutCmdLivr = coutCmdConf / (tauxLiv || 1);
    const retourCmd   = retour * (1 - tauxLiv) / (tauxLiv || 1);
    const cmdLivrees  = nbCmd * confirm * tauxLiv;
    const beneficeNet = vente - achat - livraison - emballage - coutCmdLivr - retourCmd - cod;
    const netDay      = beneficeNet * cmdLivrees;
    const netMonth    = netDay * 30;
    const totalInvesti= achat + emballage + livraison + coutCmdLivr;
    const roi         = totalInvesti > 0 ? (beneficeNet / totalInvesti) * 100 : 0;
    const roas        = pubParCmd > 0 ? vente / pubParCmd : 0;

    const fmt = (n) => 'DZD ' + n.toLocaleString('fr-FR', {minimumFractionDigits:2, maximumFractionDigits:2});

    document.getElementById('r-net').textContent     = fmt(beneficeNet);
    document.getElementById('r-daily').textContent   = fmt(netDay) + ' / Day';
    document.getElementById('r-monthly').textContent = fmt(netMonth);
    document.getElementById('r-roi').textContent     = roi.toFixed(2) + '%';

    document.getElementById('t-achat').textContent     = fmt(achat);
    document.getElementById('t-vente').textContent     = fmt(vente);
    document.getElementById('t-diff-liv').textContent  = fmt(livraison);
    document.getElementById('t-emb').textContent       = fmt(emballage);
    document.getElementById('t-cout-conf').textContent = fmt(coutCmdConf);
    document.getElementById('t-cout-livr').textContent = fmt(coutCmdLivr);
    document.getElementById('t-retour').textContent    = fmt(retourCmd);
    document.getElementById('t-cod-cmd').textContent   = fmt(cod);
    document.getElementById('t-benefice').textContent  = fmt(beneficeNet);
    document.getElementById('t-roi').textContent       = roi.toFixed(2) + '%';
    document.getElementById('t-roas').textContent      = roas.toFixed(2);

    document.getElementById('tp-cmd-day').textContent   = Math.round(nbCmd);
    document.getElementById('tp-liv-day').textContent   = cmdLivrees.toFixed(2);
    document.getElementById('tp-net-day').textContent   = fmt(netDay);
    document.getElementById('tp-net-month').textContent = fmt(netMonth);
    document.getElementById('tp-pub').textContent       = '$' + dollarDay.toFixed(2);
  }

  // ── Save button ──
  function saveAndCalculate() {
    readForm();           // sauvegarde dans le tableau simulations[]
    calculate();          // recalcule et affiche
    renderCards();        // met à jour la liste
    // Feedback visuel
    const btn = document.querySelector('.btn-primary');
    const orig = btn.textContent;
    btn.textContent = '✅ Enregistré !';
    btn.style.background = '#16a34a';
    setTimeout(() => { btn.textContent = orig; btn.style.background = ''; }, 1800);
    document.getElementById('result-section').scrollIntoView({ behavior: 'smooth' });
  }

  // ── Delete simulation ──
  function deleteSim(id) {
    if (!confirm('Supprimer cette simulation ?')) return;
    simulations = simulations.filter(s => s.id !== id);
    if (currentSimId === id) currentSimId = simulations[0]?.id || '';
    renderCards();
  }

  // ── Modal ──
  function openModal()  { document.getElementById('modal-overlay').classList.add('open'); }
  function closeModal() { document.getElementById('modal-overlay').classList.remove('open'); }

  function createSim() {
    const name = document.getElementById('new-name').value.trim().toUpperCase();
    const type = document.getElementById('new-type').value;
    if (!name) { alert('Veuillez entrer un nom'); return; }
    const id = 'sim-' + Date.now();
    simulations.unshift({
      id, name, type,
      achat:0, vente:0, dollar:250, dollarDay:0,
      cout:0, nbCmd:0, livraison:0, cod:0, emballage:0, retour:0,
      confirm:80, tauxLiv:60
    });
    renderCards();
    closeModal();
    document.getElementById('new-name').value = '';
    openDetail(id);
  }

  // ── Search / Filter ──
  function filterCards() {
    const q = document.getElementById('searchInput').value.toLowerCase();
    document.querySelectorAll('.sim-card').forEach(c => {
      const name = c.querySelector('h3').textContent.toLowerCase();
      c.style.display = name.includes(q) ? '' : 'none';
    });
  }

  // ── Override showPage to refresh cards on return ──
  const _origShowPage = showPage;
  function showPage(name) {
    _origShowPage(name);
    if (name === 'sims') renderCards();
  }

  // Init
  renderCards();
  calculate();
